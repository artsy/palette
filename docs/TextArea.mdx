## With error

<Playground>
  <Toggle initial={true}>
    {({ on, toggle }) => (
      <>
        <TextArea

        />
        <Spacer mb={3} />

          <Checkbox selected={on} onSelect={toggle} />
          <Serif size="3">Show error</Serif>
        </Flex>
      </>
    )}
  </Toggle>
</Playground>
