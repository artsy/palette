## Example

```js
import { Grid, Row, Col } from "@artsy/palette"

const App = () => {
  return (
    <Grid fluid>
      <Row>
        <Col sm={6}>
          Hello
        </Col>
        <Col sm={6}>
          How are you
        </Col>
      </Row>
    <Row>
  )
}
```
