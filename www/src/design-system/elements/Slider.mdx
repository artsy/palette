## With Label

<Playground>
  <Value initial={[0, 100]}>
    {({value, set}) => {
      const valueToString = ([min, max]) => "$" + min + " - $" + max

      return (
        <div>
          <Serif size="3" color="black60" my={2}>
            {valueToString(value)}
          </Serif>
          <Slider
            allowCross={false}
            min={10}
            max={100}
            step={1}
            defaultValue={[10, 100]}
            onChange={set}
          />
        </div>
      )
    }}

  </Value>
</Playground>
