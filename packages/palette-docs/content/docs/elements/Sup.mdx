---
name: Sup
source: elements/Sup
---

Used to display super script text within some other `Text` element. Useful for
counts.

<Playground>
  <>
    <Text variant="xxl" mb={2}>
      Lorem Ipsum <Sup color="brand">123</Sup>
    </Text>
    <Text variant="xl" mb={2}>
      Lorem Ipsum <Sup color="brand">123</Sup>
    </Text>
    <Text variant="lg-display" mb={2}>
      Lorem Ipsum <Sup color="brand">123</Sup>
    </Text>
    <Text variant="sm-display" mb={2}>
      Lorem Ipsum <Sup color="brand">123</Sup>
    </Text>
    <Text variant="sm" mb={2}>
      Lorem Ipsum <Sup color="brand">123</Sup>
    </Text>
    <Text variant="xs" mb={2}>
      Lorem Ipsum <Sup color="brand">123</Sup>
    </Text>
  </>
</Playground>
