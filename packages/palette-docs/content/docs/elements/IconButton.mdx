`IconButton` is a `Button` with a slot for an `icon` element.

<Playground>
  <>
    <IconButton
      size="small"
      variant="secondaryOutline"
      icon={<BellIcon fill="currentColor" />}
      mr={1}
    >
      Create an Alert
    </IconButton>
    <IconButton
      width="auto"
      variant="primaryBlack"
      icon={<BellIcon fill="currentColor" />}
    >
      Create an Alert
    </IconButton>
  </>
</Playground>
