    placement="bottom"

      <Text variant="xs" width={300}>

    {({ onVisible, anchorRef }) => {
      return (
        <Box textAlign="center">

            ref={anchorRef}
            variant="secondaryOutline"
            size="small"
            onClick={onVisible}

            Click to display popover
          </Button>
        </Box>
      )
    }}
  </Popover>
